CREATE TABLE member (
	_id BIGINT UNSIGNED AUTO_INCREMENT,
	login_method INT(4) NOT NULL,
	uid VARCHAR(64) NOT NULL,
	nickname VARCHAR(12) NOT NULL,
	gender INT NOT NULL,
	cash BIGINT NOT NULL,
	hair INT DEFAULT 0,
	face INT DEFAULT 0,
	bottom INT DEFAULT 0,
	shoes INT DEFAULT 0,
	skin INT DEFAULT 0,
	top INT DEFAULT 0,
	hairac INT DEFAULT -1,
	floor INT DEFAULT 0,
	wall INT DEFAULT 0,
	email VARCHAR(100),
	profile VARCHAR(50),
	language INT NOT NULL,
	recent_date DATETIME,
	fashion_date DATETIME,
	block_date DATETIME,
	PRIMARY KEY(_id)
);

CREATE TABLE closet (
	nickname VARCHAR(12) NOT NULL,
	cloth_type INT(4) NOT NULL,
	cloth_num INT(10) NOT NULL,
	CONSTRAINT UC_closet UNIQUE (nickname, cloth_type, cloth_num)
);

CREATE TABLE house (
	nickname VARCHAR(12) NOT NULL,
	furniture_type INT(4) NOT NULL,
	furniture_num INT(10) NOT NULL,
	furniture_loc VARCHAR(20),
	CONSTRAINT UC_house UNIQUE (nickname, furniture_type, furniture_num)
);

CREATE TABLE floorNwall (
	nickname VARCHAR(12) NOT NULL,
	furniture_type INT(4) NOT NULL,
	furniture_num INT(10) NOT NULL,
	CONSTRAINT UC_floorNwall UNIQUE (nickname, furniture_type, furniture_num)
);

CREATE TABLE relationship (
	type INT(4) NOT NULL,
	nickname1 VARCHAR(64) NOT NULL,
	nickname2 VARCHAR(64) NOT NULL,
	CONSTRAINT UC_block UNIQUE (type, nickname1, nickname2)
);

CREATE TABLE report (
	_id BIGINT UNSIGNED AUTO_INCREMENT,
	nickname VARCHAR(64) NOT NULL,
	contents VARCHAR(1000) NOT NULL,
	screenshot VARCHAR(100) NOT NULL,
	timestamp DATETIME,
	PRIMARY KEY(_id)
);

CREATE TABLE security_record (
	_id BIGINT UNSIGNED AUTO_INCREMENT,
	jobj VARCHAR(5000) NOT NULL,
	time DATETIME NOT NULL,
	PRIMARY KEY(_id)
);

CREATE TABLE blacklist(
	_id BIGINT UNSIGNED AUTO_INCREMENT,
	nickname VARCHAR(12) NOT NULL,
	count INT DEFAULT 1,
	PRIMARY KEY(_id)
);

	_id BIGINT UNSIGNED AUTO_INCREMENT,
	isUpdate INT DEFAULT 0,
	PRIMARY KEY(_id)

CREATE TABLE receipt (
	orderId VARCHAR(100) NOT NULL,
	nickname VARCHAR(12) NOT NULL,
	login_os INT(4) NOT NULL,
	data JSON NOT NULL,
	CONSTRAINT UC_receipt UNIQUE (orderId, login_os)
);

CREATE TABLE flower (
	nickname VARCHAR(12) NOT NULL,
	flowerNum INT NOT NULL,
	respawn_time DATETIME,
	CONSTRAINT UC_flower UNIQUE (nickname, flowerNum)
);

CREATE TABLE slot(
	nickname VARCHAR(12) NOT NULL,
	slotNum INT NOT NULL,
	hair INT DEFAULT -1,
	face INT DEFAULT 0,
	bottom INT DEFAULT -1,
	shoes INT DEFAULT -1,
	skin INT DEFAULT 0,
	top INT DEFAULT -1,
	hairac INT DEFAULT -1,
	CONSTRAINT UC_slot UNIQUE (nickname, slotNum)
);

CREATE TABLE diary(
	_id BIGINT UNSIGNED AUTO_INCREMENT,
	owner VARCHAR(12) NOT NULL,
	writer VARCHAR(12) NOT NULL,
	title VARCHAR(30),
	content VARCHAR(200) NOT NULL,
	access_level INT DEFAULT 3,
	write_date DATETIME,
	reply JSON,
	PRIMARY KEY(_id)
);

CREATE TABLE fountain(
	_id BIGINT UNSIGNED AUTO_INCREMENT,
	nickname VARCHAR(12) NOT NULL,
	cash BIGINT NOT NULL,
	PRIMARY KEY(_id)

CREATE TABLE adopt(
	nickname VARCHAR(12) NOT NULL,
	adoptNum INT NOT NULL,
	adoptData JSON,
	stamina INT DEFAULT 1,
	speed INT DEFAULT 1,
	win_count BIGINT DEFAULT 0,
	game_count BIGINT DEFAULT 0,
	CONSTRAINT UC_adopt UNIQUE (nickname, adoptNum)
